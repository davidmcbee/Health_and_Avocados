-- Join the tables all_prices and all_prod
SELECT ap.geography,
      	 ap.date,
     	 ap.type,
     	 ap.avg_price,
	 ap.total_volume,
     	 ap.units_4046,
     	 ap.units_4225,
	 ap.units_4770_,
	 ap.total_bags,
     	 ap.s_bags,
     	 ap.l_bags,
	 ap.xl_bags,
	 aprod.year_month,
	 aprod.status,
	 aprod.total_volume_produces,
	 aprod.california,
	 aprod.chile,
	 aprod.mexico,
	 aprod.peru,
	 aprod.columbia
INTO all_prices_prod
FROM all_prices as ap
LEFT JOIN all_prod as aprod
ON (ap.date = aprod.date);
